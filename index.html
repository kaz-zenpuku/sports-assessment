        function calculateResults() {

                "課題設定能力": 0,

                "課題解決能力": 0,

            const categoryCounts = {

                "課題解決能力": 0,

                "協働能力": 0,

            };


            let allAnswered = true;

            for (let i = 0; i < questions.length; i++) {

                if (!formData.has(`q${i}`)) {

                    allAnswered = false;

                    break;

                }

            }


            if (!allAnswered) {

                alert("すべての質問に回答してください。");

                return;

            }



            questions.forEach((q, index) => {

                const value = parseInt(formData.get(`q${index}`));

                if (!isNaN(value)) {

                    scores[q.category] += value;

                    categoryCounts[q.category]++;

                }

            });


            const resultsArea = document.getElementById('resultsArea');

            const scoresDiv = document.getElementById('scores');

            const overallScoreDiv = document.getElementById('overallScore');

            scoresDiv.innerHTML = '';

            overallScoreDiv.innerHTML = '';


            let totalScore = 0;

            const categoryScoresForChart = [];


            for (const category in scores) {

                const score = scores[category];

                totalScore += score;

                categoryScoresForChart.push(score);

                const evaluation = getCategoryEvaluation(score);

                scoresDiv.innerHTML += `<p><strong>${category}:</strong> ${score}点 (${evaluation})</p>`;

        function getCategoryEvaluation(score) {

            if (score >= 40) return "非常に高いレベル";

            if (score >= 30) return "良好なレベル";

            if (score >= 20) return "基本的なレベル";

            return "発展途上のレベル";

        }


        function getOverallEvaluation(score) {

            if (score >= 200) return "非常に高い探究能力";

            const imgData = canvas.toDataURL('image/png');


            const imgProps= pdf.getImageProperties(imgData);

            const pdfWidth = pdf.internal.pageSize.getWidth();

            const pdfHeight = (imgProps.height * pdfWidth) / imgProps.width;

            let position = 25; // タイトルの下に配置


            pdf.addImage(imgData, 'PNG', 10, position, pdfWidth - 20, pdfHeight -10); //マージンを考慮


            // 注意書き

            const noteY = position + pdfHeight -5; // 画像の下に配置

            if (noteY < pdf.internal.pageSize.getHeight() - 20) { // ページ下部に余裕があれば

                pdf.setFontSize(10);

                pdf.setTextColor(100);

            }



            pdf.save('sports_assessment_results.pdf');

        }
